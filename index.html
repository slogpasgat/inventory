-- 1. TABEL DATA SATUAN JAJARAN CREATE TABLE jajaran ( id INT AUTO_INCREMENT PRIMARY KEY, nama_jajaran VARCHAR(100) NOT NULL UNIQUE, kode_satuan VARCHAR(10) NULL ); -- 2. TABEL PENGGUNA (USERS & AUTH) CREATE TABLE users ( id INT AUTO_INCREMENT PRIMARY KEY, username VARCHAR(50) NOT NULL UNIQUE, password_hash VARCHAR(255) NOT NULL, role ENUM('SLOG KORPASGAT', 'JAJARAN SATUAN') NOT NULL, jajaran_id INT NULL, -- NULL jika role adalah SLOG KORPASGAT FOREIGN KEY (jajaran_id) REFERENCES jajaran(id) ON DELETE SET NULL ); -- 3. TABEL UTAMA BARANG (MASTER ITEM) CREATE TABLE master_items ( id INT AUTO_INCREMENT PRIMARY KEY, base_kategori ENUM('SENMU', 'ALPALSUS', 'RANMOR') NOT NULL, kode VARCHAR(50) UNIQUE NULL, nama_barang VARCHAR(150) NOT NULL, sub_nama_barang VARCHAR(150) NULL ); -- 4. TABEL STOK INVENTORY PER SATUAN (SENMU & ALPALSUS) CREATE TABLE inventory_base ( id INT AUTO_INCREMENT PRIMARY KEY, jajaran_id INT NOT NULL, item_id INT NOT NULL, sn_pn VARCHAR(100) NULL, -- Khusus ALPALSUS kaliber VARCHAR(50) NULL, -- Khusus SENMU satuan VARCHAR(30) NOT NULL, jumlah_awal INT DEFAULT 0, masuk INT DEFAULT 0, keluar INT DEFAULT 0, jumlah_akhir INT GENERATED ALWAYS AS (jumlah_awal + masuk - keluar) STORED, kondisi_b INT DEFAULT 0, kondisi_rr INT DEFAULT 0, kondisi_rb INT DEFAULT 0, keterangan TEXT NULL, FOREIGN KEY (jajaran_id) REFERENCES jajaran(id) ON DELETE CASCADE, FOREIGN KEY (item_id) REFERENCES master_items(id) ON DELETE CASCADE ); -- 5. TABEL DETAIL INVENTORY (KHUSUS RANMOR) CREATE TABLE inventory_ranmor ( id INT AUTO_INCREMENT PRIMARY KEY, jajaran_id INT NOT NULL, no_reg VARCHAR(50) NULL, no_lama VARCHAR(50) NULL, no_baru VARCHAR(50) NULL, merk_tipe VARCHAR(100) NOT NULL, jenis VARCHAR(50) NULL, tahun INT NULL, no_rangka VARCHAR(100) NULL, no_mesin VARCHAR(100) NULL, kondisi_b INT DEFAULT 0, kondisi_rr INT DEFAULT 0, kondisi_rb INT DEFAULT 0, kondisi_persen DECIMAL(5,2) NULL, cc INT NULL, bbm VARCHAR(30) NULL, warna VARCHAR(30) NULL, hak_pakai VARCHAR(100) NULL, masa_kbnkb DATE NULL, proses VARCHAR(100) NULL, maju VARCHAR(50) NULL, turun VARCHAR(50) NULL, ukuran VARCHAR(50) NULL, ban INT NULL, aki INT NULL, keterangan TEXT NULL, FOREIGN KEY (jajaran_id) REFERENCES jajaran(id) ON DELETE CASCADE ); -- 6. TABEL TRANSAKSI MUTASI / ALOKASI BARANG CREATE TABLE alokasi_logistik ( id INT AUTO_INCREMENT PRIMARY KEY, jajaran_asal_id INT NOT NULL, jajaran_tujuan_id INT NOT NULL, base_kategori ENUM('SENMU', 'ALPALSUS', 'RANMOR') NOT NULL, item_id INT NULL, -- Hubungan ke master_items jika SENMU/ALPALSUS ranmor_id INT NULL, -- Hubungan ke inventory_ranmor jika RANMOR jumlah_dialokasikan INT NOT NULL, status ENUM('PENDING', 'APPROVED', 'REJECTED') DEFAULT 'PENDING', notifikasi_dibaca TINYINT(1) DEFAULT 0, tanggal_pengajuan TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (jajaran_asal_id) REFERENCES jajaran(id), FOREIGN KEY (jajaran_tujuan_id) REFERENCES jajaran(id), FOREIGN KEY (item_id) REFERENCES master_items(id), FOREIGN KEY (ranmor_id) REFERENCES inventory_ranmor(id) );
